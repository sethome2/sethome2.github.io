线性素数筛

bool is_prime[1000001] = {};

void produce()
{
  memset(&is_prime, 1, sizeof(is_prime));
  is_prime[0] = is_prime[1] = 0;
  for (unsigned long long i = 2; i <= 1000000; i++)
  {
    if (i * i <= 1000000)
      for (unsigned long long j = i * i; j <= 1000000; j += i)
        is_prime[j] = 0;
  }
}
